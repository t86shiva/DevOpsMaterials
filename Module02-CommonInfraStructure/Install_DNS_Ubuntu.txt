# Restart bind
sudo /etc/init.d/bind9 start
# Check configuration... Verify....

##### Edit Local File...
sudo vim /etc/bind/named.conf.local
zone "edurekademo.com" {
        type master;

##Create Forward Zone files
sudo mkdir /etc/bind/zones
cd /etc/bind/zones
sudo cp ../db.local ./db.edurekademo.com
sudo vi /etc/bind/zones/db.edurekademo.com

$TTL    604800
@       IN      SOA     demo1.edurekademo.com. admin.edurekademo.com. (
                              2         ; Serial
                         604800         ; Refresh
                          86400         ; Retry
                        2419200         ; Expire
                         604800 )       ; Negative Cache TTL
;
#@       IN      NS      localhost.      ; delete this line
#@       IN      A       127.0.0.1       ; delete this line
#@       IN      AAAA    ::1             ; delete this line
; name servers - NS records
    IN      NS      demo1.edurekademo.com.

; name servers - A records
demo1.edurekademo.com.          IN      A       192.168.33.80

; 10.128.0.0/16 - A records
chef.edurekademo.com.        IN      A      192.168.33.80
demo2.edurekademo.com.        IN      A      192.168.33.81
proxy.edurekademo.com.        IN      A      192.168.33.82
chef.edurekademo.com.        IN      A      192.168.33.83

##Create Reverse Zone files... 
sudo vi /etc/bind/zones/db.192.168 # must match what you have in named.local.conf
$TTL    604800
@       IN      SOA     demo1.edurekademo.com. admin.edurekademo.com. (
                              3         ; Serial
                         604800         ; Refresh
                          86400         ; Retry
                        2419200         ; Expire
                         604800 )       ; Negative Cache TTL
; name servers
      IN      NS      demo1.edurekademo.com.

; PTR Records
80.33   IN      PTR     demo1.edurekademo.com.    ; 192.168.33.80
81.33   IN      PTR     demo2.edurekademo.com.    ; 192.168.33.81
82.33   IN      PTR     proxy.edurekademo.com.    ; 192.168.33.82
83.33   IN      PTR     chef.edurekademo.com.    ; 192.168.33.83

## Check Configuration.
